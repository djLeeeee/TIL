```python
# 16901 XOR MST
from sys import stdin

input = stdin.readline


def sol(array, bit):
    global ans
    if array and bit >= 0:
        left = []
        right = []
        for num in array:
            if num & (1 << bit):
                right.append(num)
            else:
                left.append(num)
        edge = None
        for l in left:
            for r in right:
                if edge is None:
                    edge = l ^ r
                else:
                    edge = min(edge, l ^ r)
        if edge:
            ans += edge
        bit -= 1
        sol(left, bit)
        sol(right, bit)


n = int(input())
ans = 0
sol(list(map(int, input().split())), 29)
print(ans)
```
